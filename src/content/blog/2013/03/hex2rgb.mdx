Debido a que ahora trabajo como maquetador ;) he andado navegando por algunas páginas web (entre ellas: http://www.javascripter.net/) por lo que necesitando convertir los colores que se utilizan en la web, pasarlos de HEX a RGB o viceversa pues encontré esta sencilla solución:<br />

<div className="flex flex-wrap gap-2">
  <div className="flex gap-2">
      <label>HEX:</label>
      <input className="border-2 rounded border-black dark:border-white px-2"
             type="text" id="hex" />
      <input
        className="border-2 rounded text-white border-black px-3 bg-gray-800 dark:bg-gray-200 dark:text-black"
        type="button" id="to" value="Convertir" />
  </div>
  <div className="flex gap-2">
    <label>R:</label><input disabled className="border-2 rounded border-black dark:border-white px-2"
              type="text" id="r" />
    <label>G:</label><input disabled className="border-2 rounded border-black dark:border-white px-2"
              type="text" id="g" />
    <label>B:</label><input disabled className="border-2 rounded border-black dark:border-white px-2"
              type="text" id="b" />
  </div>
</div>

Si desean el código para realizar esta "increíble hazaña":<br />

```js
function hexToR(h) {
  return parseInt((cutHex(h)).substring(0,2),16);
}

function hexToG(h) {
  return parseInt((cutHex(h)).substring(2,4),16);
}

function hexToB(h) {
  return parseInt((cutHex(h)).substring(4,6),16);
}

function cutHex(h) {
  return (h.charAt(0)==="#") ? h.substring(1,7) : h;
}
```


<script>{`
    const $ = document.querySelector.bind(document);
    $('#to').addEventListener('click', function(){
        var hex = $('#hex').value;
        $('#r').value = (hexToR(hex));
        $('#g').value = (hexToG(hex));
        $('#b').value = (hexToB(hex));
    });

    function hexToR(h) {
         return parseInt((cutHex(h)).substring(0,2),16);
    }

    function hexToG(h) {
         return parseInt((cutHex(h)).substring(2,4),16);
    }

    function hexToB(h) {
         return parseInt((cutHex(h)).substring(4,6),16);
    }

    function cutHex(h) {
         return (h.charAt(0)=="#") ? h.substring(1,7) : h;
    }
`}</script>
